41 void Engine::validateRetCode(hebench::APIBridge::ErrorCode err_code, bool last_error) const
42 {
43  if (err_code != HEBENCH_ECODE_SUCCESS)
44  {
45  std::stringstream ss;
46  ss << getErrorDescription(err_code);
47  if (last_error)
48  {
49  std::string s = getLastErrorDescription();
50  if (!s.empty())
51  ss << std::endl
52  << s;
53  } // end if
54  throw hebench::Common::ErrorException(ss.str(), err_code);
55  } // end if

70 void Engine::init()
71 {
72  std::stringstream ss;
73 
74  // initialize backend engine
75  hebench::APIBridge::ErrorCode err_code = hebench::APIBridge::initEngine(&m_handle);
76  if (err_code != HEBENCH_ECODE_SUCCESS)
77  {
78  try
79  {
80  std::string s = getErrorDescription(err_code);
81  ss = std::stringstream(s);
82  ss << s;
83  s = getLastErrorDescription();
84  if (!s.empty())
85  ss << std::endl
86  << s;
87  }
88  catch (...)
89  {
90  ss = std::stringstream(IL_LOG_MSG_CLASS("Error initializing backend engine."));
91  }
92  throw hebench::Common::ErrorException(ss.str(), err_code);
93  } // end if
94 
95  // register benchmarks from backend
96  std::uint64_t count;
97  validateRetCode(hebench::APIBridge::subscribeBenchmarksCount(m_handle, &count));
98  m_h_bench_desc.resize(count);
99  validateRetCode(hebench::APIBridge::subscribeBenchmarks(m_handle, m_h_bench_desc.data()));
100 }

194 std::vector<std::vector<hebench::APIBridge::WorkloadParam>>
195 Engine::getDefaultWorkloadParams(std::size_t index) const
196 {
197  if (index >= m_h_bench_desc.size())

199 
200  std::uint64_t param_count, default_count;
201  validateRetCode(hebench::APIBridge::getWorkloadParamsDetails(m_handle,
202  m_h_bench_desc[index],
203  &param_count,
204  &default_count));
205  std::vector<std::vector<hebench::APIBridge::WorkloadParam>> retval(default_count);
206  if (!retval.empty())
207  {
208  std::vector<hebench::APIBridge::WorkloadParams> wps(retval.size());
209  for (std::size_t i = 0; i < retval.size(); ++i)
210  {
211  retval[i].resize(param_count);
212  wps[i].count = retval[i].size();
213  wps[i].params = retval[i].data();
214  } // end for
215  hebench::APIBridge::BenchmarkDescriptor tmp_bd;
216  validateRetCode(hebench::APIBridge::describeBenchmark(m_handle,
217  m_h_bench_desc[index],
218  &tmp_bd,
219  wps.data()));
220  } // end if
