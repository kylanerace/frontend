56 class ConfigImporterImpl
57 {
58 public:
59  static std::vector<BenchmarkRequest> importYAML2BenchmarkRequest(std::size_t benchmark_index,
60  const YAML::Node &yaml_bench,
61  const hebench::TestHarness::Engine &engine,
62  std::uint64_t fallback_min_test_time,
63  std::uint64_t fallback_sample_size);
64 
65 private:
66  template <typename T>
67  static std::size_t computeComponentSize(const YAML::Node &node_from,
68  const YAML::Node &node_to,
69  const YAML::Node &node_step)
70  {
71  T param_step = node_step.as<T>();
72  if (param_step == static_cast<T>(0))
73  param_step = std::numeric_limits<T>::max();
74  T tmp_to = node_to.as<T>();
75  T tmp_from = node_from.as<T>();
76  if (tmp_to < tmp_from)
77  tmp_to = tmp_from;
78  return static_cast<std::size_t>((tmp_to - tmp_from) / param_step) + 1;
79  }
80 
81  template <typename T>
82  static T computeParamValue(std::size_t count,
83  const YAML::Node &node_from,
84  const YAML::Node &node_step)
85  {
86  return static_cast<T>(count * node_step.as<T>() + node_from.as<T>());
87  }
88 };

94 std::vector<BenchmarkRequest> ConfigImporterImpl::importYAML2BenchmarkRequest(std::size_t benchmark_index,
95  const YAML::Node &yaml_bench,
96  const TestHarness::Engine &engine,
97  std::uint64_t fallback_min_test_time,
98  std::uint64_t fallback_sample_size)
99 {
100  assert(yaml_bench["ID"].as<decltype(benchmark_index)>() == benchmark_index);
101 
102  std::vector<BenchmarkRequest> retval;
103 
104  std::uint64_t default_min_test_time_ms = 0;
105  std::vector<std::uint64_t> default_sample_sizes;
106 
107  if (yaml_bench["default_min_test_time"].IsDefined())
108  default_min_test_time_ms = yaml_bench["default_min_test_time"].as<decltype(default_min_test_time_ms)>();
109 
110  if (yaml_bench["default_sample_sizes"].IsDefined())
111  {
112  if (!yaml_bench["default_sample_sizes"].IsMap())
113  {
114  YAML::Mark yaml_mark = yaml_bench["default_sample_sizes"].Mark();
115  std::stringstream ss;
116  ss << "Value of field `default_sample_sizes` in ";
117  if (yaml_mark.is_null())
118  ss << "benchmark with ID " << benchmark_index;
119  else
120  ss << "line " << yaml_mark.line + 1;
121  ss << " must be a map matching operation parameter index to the corresponding sample size.";
122  throw std::runtime_error(ss.str());
123  } // end if
124  for (auto it = yaml_bench["default_sample_sizes"].begin(); it != yaml_bench["default_sample_sizes"].end(); ++it)
125  {
126  // retrieve the sample size in format `index: size`
127  // this way, users can specify sample sizes for some, but not necessarily all operation parameters.
128  std::size_t key = it->first.as<std::size_t>();
129  if (default_sample_sizes.size() <= key)
130  default_sample_sizes.resize(key + 1, 0);
131  default_sample_sizes[key] = it->second.as<std::uint64_t>();
132  } // end for
133  } // end if
134 
135  if (yaml_bench["params"].IsDefined() && yaml_bench["params"].size() > 0)
136  {
137  // create a component counter to iterate over the parameters using from-to-step model
138  std::vector<std::size_t> component_sizes(yaml_bench["params"].size());
139  for (std::size_t param_i = 0; param_i < component_sizes.size(); ++param_i)
140  {
141  // validate node
142 
143  if (!yaml_bench["params"][param_i].IsDefined())
144  {
145  std::stringstream ss;
146  ss << "workload parameter index " << param_i << " not found in benchmark with ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
147  throw std::runtime_error(ss.str());
148  } // end if
149  if (!yaml_bench["params"][param_i]["name"].IsDefined())
150  {
151  std::stringstream ss;
152  ss << "Field \"name\" not found for workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
153  throw std::runtime_error(ss.str());
154  } // end if
155  if (!yaml_bench["params"][param_i]["type"].IsDefined())
156  {
157  std::stringstream ss;
158  ss << "Field \"type\" not found for workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
159  throw std::runtime_error(ss.str());
160  } // end if
161  if (!yaml_bench["params"][param_i]["value"].IsDefined())
162  {
163  std::stringstream ss;
164  ss << "Field \"value\" not found for workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
165  throw std::runtime_error(ss.str());
166  } // end if
167 
168  YAML::Node yaml_param_value = yaml_bench["params"][param_i]["value"];
169  if (!yaml_param_value["from"].IsDefined())
170  {
171  std::stringstream ss;
172  ss << "Field \"from\" not found for \"value\" in workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
173  throw std::runtime_error(ss.str());
174  } // end if
175  if (!yaml_param_value["to"].IsDefined())
176  {
177  std::stringstream ss;
178  ss << "Field \"to\" not found for \"value\" in workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
179  throw std::runtime_error(ss.str());
180  } // end if
181  if (!yaml_param_value["step"].IsDefined())
182  {
183  std::stringstream ss;
184  ss << "Field \"step\" not found for \"value\" in workload parameter index " << param_i << ", benchmark ID " << benchmark_index << ", defined at line " << yaml_bench.Mark().line + 1 << ".";
185  throw std::runtime_error(ss.str());
186  } // end if
187 
188  // convert param from-to-step into counter
189  std::string param_type = yaml_bench["params"][param_i]["type"].as<std::string>();
190  std::transform(param_type.begin(), param_type.end(), param_type.begin(),
191  [](unsigned char c) { return std::tolower(c); });
192  if (param_type == "uint64")
193  {
194  component_sizes[param_i] =
195  computeComponentSize<std::uint64_t>(yaml_param_value["from"],
196  yaml_param_value["to"],
197  yaml_param_value["step"]);
198  } // end if
199  else if (param_type == "int64")
200  {
201  component_sizes[param_i] =
202  computeComponentSize<std::int64_t>(yaml_param_value["from"],
203  yaml_param_value["to"],
204  yaml_param_value["step"]);
205  } // end if
206  else if (param_type == "float64")
207  {
208  component_sizes[param_i] =
209  computeComponentSize<double>(yaml_param_value["from"],
210  yaml_param_value["to"],
211  yaml_param_value["step"]);
212  } // end if
213  else
214  {
215  std::stringstream ss;
216  ss << "Invalid \"type\" found in parameter index " << param_i
217  << ", benchmark ID " << benchmark_index << ": "
218  << yaml_bench["params"][param_i]["type"].as<std::string>() << ".";
219  throw std::runtime_error(ss.str());
220  } // end if
221  } // end for
222 
223  hebench::Utilities::Math::ComponentCounter c_counter(component_sizes);
224 
225  do
226  {
227  const std::vector<std::size_t> &count = c_counter.getCount();
228  hebench::TestHarness::BenchmarkDescription::Configuration config;
229  config.default_min_test_time_ms = default_min_test_time_ms <= 0 ? fallback_min_test_time : default_min_test_time_ms;
230  config.fallback_default_sample_size = fallback_sample_size;
231  config.default_sample_sizes = default_sample_sizes;
232  for (std::size_t param_i = 0; param_i < count.size(); ++param_i)
233  {
234  // fill out WorkloadParam struct
235  hebench::APIBridge::WorkloadParam w_param;
236  YAML::Node yaml_param = yaml_bench["params"][param_i];
237  YAML::Node yaml_param_value = yaml_param["value"];
238  std::string param_type = yaml_param["type"].as<std::string>();
239  std::transform(param_type.begin(), param_type.end(), param_type.begin(),
240  [](unsigned char c) { return std::tolower(c); });
241 
242  if (param_type == "uint64")
243  {
244  w_param.data_type = hebench::APIBridge::WorkloadParamType::UInt64;
245  w_param.u_param = computeParamValue<decltype(w_param.u_param)>(count[param_i],
246  yaml_param_value["from"],
247  yaml_param_value["step"]);
248  } // end if
249  else if (param_type == "int64")
250  {
251  w_param.data_type = hebench::APIBridge::WorkloadParamType::Int64;
252  w_param.i_param = computeParamValue<decltype(w_param.i_param)>(count[param_i],
253  yaml_param_value["from"],
254  yaml_param_value["step"]);
255  } // end if
256  else if (param_type == "float64")
257  {
258  w_param.data_type = hebench::APIBridge::WorkloadParamType::Float64;
259  w_param.f_param = computeParamValue<decltype(w_param.f_param)>(count[param_i],
260  yaml_param_value["from"],
261  yaml_param_value["step"]);
262  } // end if
263 
264  hebench::Utilities::copyString(w_param.name, HEBENCH_MAX_BUFFER_SIZE, yaml_param["name"].as<std::string>());
265  config.w_params.push_back(w_param);
266  } // end for
267 
268  // check if benchmark is supported with specified parameters
269  try
270  {
271  hebench::TestHarness::IBenchmarkDescriptor::DescriptionToken::Ptr p_token =
272  engine.describeBenchmark(benchmark_index, config);
273  if (!p_token)
274  throw std::runtime_error("Unable to match benchmark.");
275  retval.emplace_back();
276  retval.back().index = benchmark_index;
277  retval.back().configuration = p_token->getBenchmarkConfiguration();
278  }
279  catch (std::runtime_error &ex)
280  {
281  std::stringstream ss;
282  ss << "Benchmark ID: " << benchmark_index << std::endl
283  << ex.what();
284  throw std::runtime_error(ss.str());
285  }
286  } while (!c_counter.inc()); // repeat until we have done all possible parameters
287  } // end if
288  else
289  {
290  // add an empty parameter list for workload
291  try
292  {
293  retval.emplace_back();
294  retval.back().configuration.default_min_test_time_ms = default_min_test_time_ms;
295  retval.back().configuration.default_sample_sizes = default_sample_sizes;
296  hebench::TestHarness::IBenchmarkDescriptor::DescriptionToken::Ptr p_token =
297  engine.describeBenchmark(benchmark_index, retval.back().configuration);
298  if (!p_token)
299  throw std::runtime_error("Unable to match benchmark.");
300  retval.back().index = benchmark_index;
301  retval.back().configuration = p_token->getBenchmarkConfiguration();
302  }
303  catch (std::runtime_error &ex)
304  {
305  std::stringstream ss;
306  ss << "Benchmark ID: " << benchmark_index << std::endl
307  << ex.what();
308  throw std::runtime_error(ss.str());
309  }
310  } // end else
311 
312  return retval;
313 }

319 YAML::Node ConfigExporterImpl::createWorkloadParamYAMLNode(const hebench::APIBridge::WorkloadParam &w_param)
320 {
321  YAML::Node retval;
322  retval["name"] = w_param.name;
323  switch (w_param.data_type)
324  {
325  case hebench::APIBridge::WorkloadParamType::Int64:
326  retval["type"] = "Int64";
327  retval["value"] = createWorkloadParamValueYAMLNode<decltype(w_param.i_param)>(w_param.i_param);
328  break;
329 
330  case hebench::APIBridge::WorkloadParamType::UInt64:
331  retval["type"] = "UInt64";
332  retval["value"] = createWorkloadParamValueYAMLNode<decltype(w_param.u_param)>(w_param.u_param);
333  break;
334 
335  case hebench::APIBridge::WorkloadParamType::Float64:
336  retval["type"] = "Float64";
337  retval["value"] = createWorkloadParamValueYAMLNode<decltype(w_param.f_param)>(w_param.f_param);
338  break;
339 
340  default:
341  throw std::invalid_argument("Invalid data type detected in 'w_param'.");
342  break;
343  } // end switch
344 
345  return retval;
346 }

348 void ConfigExporterImpl::exportBenchmarkRequest2YAML(YAML::Emitter &out,
349  const BenchmarkRequest &bench_req,
350  const hebench::TestHarness::BenchmarkDescription::Description &description)
351 {
352  const hebench::TestHarness::BenchmarkDescription::Configuration &config = bench_req.configuration;
353  std::filesystem::path path_bench_desc = description.path;
354  std::stringstream ss;
355  auto path_it = path_bench_desc.begin();
356  if (path_it == path_bench_desc.end())
357  throw std::logic_error("Invalid benchmark description text.");
358  bool b_first_it = true;
359  for (++path_it; path_it != path_bench_desc.end(); ++path_it)
360  {
361  if (b_first_it)
362  b_first_it = false;
363  else
364  ss << " | ";
365  std::string s_tmp = *path_it;
366  s_tmp.erase(0, s_tmp.find_first_not_of('\"'));
367  s_tmp.erase(s_tmp.find_last_not_of('\"') + 1);
368  ss << s_tmp;
369  } // end for
370 
371  YAML::Node node_benchmark;
372  out << YAML::Newline
373  << YAML::Comment("Benchmark with workload parameters:") << YAML::Newline
374  << YAML::Comment(" " + description.workload_name) << YAML::Newline
375  << YAML::Comment("Descriptor:") << YAML::Newline
376  << YAML::Comment(" " + ss.str());
377  node_benchmark["ID"] = bench_req.index;
378  node_benchmark["default_min_test_time"] = config.default_min_test_time_ms;
379  if (!config.default_sample_sizes.empty())
380  {
381  for (std::size_t op_param_i = 0; op_param_i < config.default_sample_sizes.size(); ++op_param_i)
382  node_benchmark["default_sample_sizes"].push_back(config.default_sample_sizes[op_param_i]);
383  node_benchmark["default_sample_sizes"]["to_map"] = "yes"; // convert sequence to map to have the indices as keys
384  node_benchmark["default_sample_sizes"].remove("to_map");
385  } // end if
386  if (!config.w_params.empty())
387  {
388  for (std::size_t param_i = 0; param_i < config.w_params.size(); ++param_i)
389  node_benchmark["params"].push_back(createWorkloadParamYAMLNode(config.w_params[param_i]));
390  node_benchmark["params"]["to_map"] = "yes"; // convert sequence to map to have the indices as keys
391  node_benchmark["params"].remove("to_map");
392  } // end if
393  out << node_benchmark << YAML::Newline;
394 }

400 BenchmarkConfigurator::BenchmarkConfigurator(std::weak_ptr<hebench::TestHarness::Engine> wp_engine,
401  const std::string s_backend) :
402  m_wp_engine(wp_engine),
403  m_s_backend(s_backend)
404 {
405  std::shared_ptr<hebench::TestHarness::Engine> p_engine = wp_engine.lock();
406  if (!p_engine)
407  throw std::invalid_argument(IL_LOG_MSG_CLASS("Invalid empty 'wp_engine'."));
408 
409  std::size_t count_benchmarks = p_engine->countBenchmarks();
410  m_default_benchmarks.reserve(count_benchmarks);
411  for (std::size_t bench_i = 0; bench_i < count_benchmarks; ++bench_i)
412  {
413  std::vector<std::vector<hebench::APIBridge::WorkloadParam>> w_params_set =
414  p_engine->getDefaultWorkloadParams(bench_i);
415  for (std::size_t w_params_i = 0; w_params_i < w_params_set.size(); ++w_params_i)
416  {
417  m_default_benchmarks.emplace_back();
418  BenchmarkRequest &bench_description = m_default_benchmarks.back();
419  bench_description.configuration.w_params = std::move(w_params_set[w_params_i]);
420  // obtain full description for the benchmark
421  hebench::TestHarness::IBenchmarkDescriptor::DescriptionToken::Ptr p_token =
422  p_engine->describeBenchmark(bench_i, bench_description.configuration);
423  if (!p_token)
424  throw std::runtime_error(IL_LOG_MSG_CLASS("Unexpected error: no matching benchmark found for backend request."));
425  // store the data needed to retrieve the benchmark
426  bench_description.index = bench_i;
427  bench_description.configuration = p_token->getBenchmarkConfiguration();
428  //bench_description.description = p_token->getDescription();
429  } // end for
430  } // end for

475  << " auto-generated benchmark with the same ID." << std::endl

477  << " for each workload parameter. Invalid values will cause the benchmark to" << std::endl

479  out << YAML::Comment(ss.str()) << YAML::Newline;
480

482  out << YAML::Newline;
483

502 
503  ss = std::stringstream();
504  ss << "Random seed to use when generating synthetic data for these benchmarks." << std::endl
505  << "Type: unsigned int. Defaults to system time when not present.";

509 
510  out << YAML::EndMap;
511 
512  // output benchmark list
513 
514  out << YAML::BeginMap;

533  } // end for
534 
535  out << YAML::EndSeq
536  << YAML::EndMap;
537 
538  hebench::Utilities::writeToFile(
539  yaml_filename,
540  [&out](std::ostream &os) {
541  os << out.c_str();
542  },
543  false, false);

546 std::vector<BenchmarkRequest> BenchmarkConfigurator::loadConfiguration(const std::string &yaml_filename,
547  std::uint64_t &random_seed) const
548 {
549  std::vector<BenchmarkRequest> retval;
550  std::uint64_t default_min_test_time_ms = 0;
551  std::uint64_t default_sample_size = 0;
552 
553  std::shared_ptr<hebench::TestHarness::Engine> p_engine = m_wp_engine.lock();
554  if (!p_engine)
