24 std::uint64_t BenchmarkDescriptorCategory::fetchVectorSize(const std::vector<hebench::APIBridge::WorkloadParam> &w_params)
25 {
26  assert(WorkloadParameterCount == 1);
27  assert(OpParameterCount == 2);
28  assert(OpResultCount == 1);
29 
30  std::uint64_t retval;
31 
32  if (w_params.size() < WorkloadParameterCount)
33  {
34  std::stringstream ss;
35  ss << "Insufficient workload parameters in 'w_params'. Expected " << WorkloadParameterCount
36  << ", but " << w_params.size() << "received.";
37  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
38  } // end if
39 
40  for (std::size_t i = 0; i < WorkloadParameterCount; ++i)
41  if (w_params[i].data_type != WorkloadParameterType[i])
42  {
43  std::stringstream ss;
44  ss << "Invalid type for workload parameter " << i
45  << ". Expected type ID " << WorkloadParameterType[i] << ", but " << w_params[i].data_type << " received.";
46  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
47  } // end if
48  else if (w_params[i].u_param <= 0)
49  {
50  std::stringstream ss;
51  ss << "Invalid number of elements for vector in workload parameter " << i
52  << ". Expected positive integer, but " << w_params[i].u_param << " received.";
53  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
54  } // end if
55 
56  retval = w_params.at(0).u_param;
57 
58  return retval;

125  {
126  std::transform(a, a + elem_count, // a[0..elem_count]
127  b, // b[0..elem_count]
128  result, // result[0..elem_count]
129  std::plus<T>()); // eltwise add

133 void DataGeneratorHelper::vectorEltwiseAdd(hebench::APIBridge::DataType data_type,
134  void *result, const void *a, const void *b,
135  std::uint64_t elem_count)
136 {
137  switch (data_type)
138  {
139  case hebench::APIBridge::DataType::Int32:
140  vectorEltwiseAdd<std::int32_t>(reinterpret_cast<std::int32_t *>(result),
141  reinterpret_cast<const std::int32_t *>(a), reinterpret_cast<const std::int32_t *>(b),
142  elem_count);
143  break;
144 
145  case hebench::APIBridge::DataType::Int64:
146  vectorEltwiseAdd<std::int64_t>(reinterpret_cast<std::int64_t *>(result),
147  reinterpret_cast<const std::int64_t *>(a), reinterpret_cast<const std::int64_t *>(b),
148  elem_count);
149  break;
150 
151  case hebench::APIBridge::DataType::Float32:
152  vectorEltwiseAdd<float>(reinterpret_cast<float *>(result),
153  reinterpret_cast<const float *>(a), reinterpret_cast<const float *>(b),
154  elem_count);
155  break;
156 
157  case hebench::APIBridge::DataType::Float64:
158  vectorEltwiseAdd<double>(reinterpret_cast<double *>(result),
159  reinterpret_cast<const double *>(a), reinterpret_cast<const double *>(b),
160  elem_count);
161  break;
162 
163  default:
164  throw std::invalid_argument(IL_LOG_MSG_CLASS("Unknown data type."));
165  break;
166  } // end switch
167 }

183 void DataGenerator::init(std::uint64_t vector_size,
184  std::uint64_t batch_size_a,
185  std::uint64_t batch_size_b,
186  hebench::APIBridge::DataType data_type)
187 {
188  // Load/generate and initialize the data for vector element-wise addition:
189  // C = A + B
190 
191  // number of samples in each input parameter and output
192  std::size_t batch_sizes[InputDim0 + OutputDim0] = {
193  batch_size_a,
194  batch_size_b,
195  batch_size_a * batch_size_b
196  };
197  // initialize base data (data packs)
198  PartialDataLoader::init(InputDim0, batch_sizes, OutputDim0);
199 
200  // compute buffer size in bytes for each vector
201  std::uint64_t buffer_sizes[InputDim0 + OutputDim0];
202  for (std::size_t i = 0; i < InputDim0 + OutputDim0; ++i)
203  {
204  buffer_sizes[i] = vector_size * PartialDataLoader::sizeOf(data_type);
205  } // end for
206 
207  // allocate memory for each vector buffer
208  allocate(buffer_sizes, // sizes (in bytes) for each input vector
209  InputDim0, // number of input vectors
210  buffer_sizes + InputDim0, // sizes (in bytes) for each output vector
211  OutputDim0); // number of output vectors
212 
213  // at this point all NativeDataBuffers have been allocated and pointed to the correct locations
214 
215  // fill up each vector data
216 
217  // input
218  for (std::size_t vector_i = 0; vector_i < InputDim0; ++vector_i)
219  {
220  for (std::uint64_t i = 0; i < batch_sizes[vector_i]; ++i)
221  {
222  // generate the data
223  DataGeneratorHelper::generateRandomVectorU(data_type,
224  getParameterData(vector_i).p_buffers[i].p,
225  vector_size, -10.0, 10.0);
226  } // end for
227  } // end for
228 
229  // output
230  //#pragma omp parallel for collapse(2)
231  for (std::uint64_t a_i = 0; a_i < batch_sizes[0]; ++a_i)
232  {
233  for (std::uint64_t b_i = 0; b_i < batch_sizes[1]; ++b_i)
234  {
235  // find the index for the result buffer based on the input indices
236  std::uint64_t ppi[] = { a_i, b_i };
237  std::uint64_t r_i = getResultIndex(ppi);
238 
239  // generate the data
240  DataGeneratorHelper::vectorEltwiseAdd(data_type,
241  getResultData(0).p_buffers[r_i].p, // C
242  getParameterData(0).p_buffers[a_i].p, // A
243  getParameterData(1).p_buffers[b_i].p, // B
244  vector_size);
245  } // end for
246  } // end for
247 
248  // all data has been generated at this point
249 }
