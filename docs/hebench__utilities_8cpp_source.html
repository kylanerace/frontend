18 {

21 
22  // convert all non-alnum to underscores and, if requested, alnum to lowercase
23  std::transform(s_tmp.begin(), s_tmp.end(), s_tmp.begin(),
24  [to_lowercase](unsigned char c) { return (std::isalnum(c) || c == '.' ?
25  (to_lowercase ? std::tolower(c) : c) :
26  '_'); });
27  // remove leading and trailing underscores
28  s_tmp.erase(0, s_tmp.find_first_not_of('_'));
29  s_tmp.erase(s_tmp.find_last_not_of('_') + 1);
30

33  while (pos < s_tmp.size())
34  {
35  std::size_t i = s_tmp.find_first_of('_', pos);
36  if (i == std::string::npos)
37  ss_retval << s_tmp.substr(pos);
38  else
39  {
40  if (i < s_tmp.size())
41  ++i;
42  ss_retval << s_tmp.substr(pos, i - pos);
43  i = s_tmp.find_first_not_of('_', i);
44  } // end if
45  pos = i;
46  } // end while
47 
48  return ss_retval.str();
