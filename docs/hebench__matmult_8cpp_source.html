26 std::array<std::pair<std::uint64_t, std::uint64_t>, BenchmarkDescriptorCategory::OpParameterCount>
27 BenchmarkDescriptorCategory::fetchMatrixSizes(const std::vector<hebench::APIBridge::WorkloadParam> &w_params)
28 {
29  assert(WorkloadParameterCount == 3);
30  assert(OpParameterCount == 2);
31  assert(OpResultCount == 1);
32 
33  std::array<std::pair<std::uint64_t, std::uint64_t>, OpParameterCount> retval;
34 
35  if (w_params.size() < WorkloadParameterCount)
36  {
37  std::stringstream ss;
38  ss << "Insufficient workload parameters in 'w_params'. Expected " << WorkloadParameterCount
39  << ", but " << w_params.size() << "received.";
40  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
41  } // end if
42 
43  // validate workload parameters
44  for (std::size_t i = 0; i < WorkloadParameterCount; ++i)
45  if (w_params[i].data_type != WorkloadParameterType[i])
46  {
47  std::stringstream ss;
48  ss << "Invalid type for workload parameter " << i
49  << ". Expected type ID " << WorkloadParameterType[i] << ", but " << w_params[i].data_type << " received.";
50  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
51  } // end if
52  else if (w_params[i].u_param <= 0)
53  {
54  std::stringstream ss;
55  ss << "Invalid matrix size in workload parameter " << i
56  << ". Expected positive integer, but " << w_params[i].u_param << " received.";
57  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
58  } // end if
59 
60  retval.at(0) = std::make_pair(w_params.at(0).u_param, w_params.at(1).u_param);
61  retval.at(1) = std::make_pair(w_params.at(1).u_param, w_params.at(2).u_param);
62 
63  return retval;
64 }

121 public:

123  void *mat_result, std::uint64_t rows, std::uint64_t cols,
124  double mean, double stddev);
125  static void matMul(hebench::APIBridge::DataType data_type,
126  void *mat_result, const void *mat_a, const void *mat_b,
127  std::uint64_t rows_a, std::uint64_t cols_a, std::uint64_t cols_b);
128 
129 protected:
130  DataGeneratorHelper() {}
131 
132 private:
133  template <class T>
134  static void matMul(T *mat_result, const T *mat_a, const T *mat_b,
135  std::uint64_t rows_a, std::uint64_t cols_a, std::uint64_t cols_b)
136  {
137  // perform matrix multiplication (straight-forward way,
138  // maybe optimize later)
139  for (std::uint64_t row_a = 0; row_a < rows_a; ++row_a)
140  for (std::uint64_t col_b = 0; col_b < cols_b; ++col_b)
141  {
142  mat_result[row_a * cols_b + col_b] = 0;
143  for (std::uint64_t col_a = 0; col_a < cols_a; ++col_a)
144  {
145  std::uint64_t row_b = col_a;
146  mat_result[row_a * cols_b + col_b] += mat_a[row_a * cols_a + col_a] * mat_b[row_b * cols_b + col_b];
147  } // end for
148  } // end for
149  }
150 };

161 void DataGeneratorHelper::matMul(hebench::APIBridge::DataType data_type,
162  void *mat_result,
163  const void *mat_a, const void *mat_b,
164  uint64_t rows_a, uint64_t cols_a, uint64_t cols_b)
165 {
166  switch (data_type)
167  {
168  case hebench::APIBridge::DataType::Int32:
169  matMul<std::int32_t>(reinterpret_cast<std::int32_t *>(mat_result),
170  reinterpret_cast<const std::int32_t *>(mat_a), reinterpret_cast<const std::int32_t *>(mat_b),
171  rows_a, cols_a, cols_b);
172  break;
173 
174  case hebench::APIBridge::DataType::Int64:
175  matMul<std::int64_t>(reinterpret_cast<std::int64_t *>(mat_result),
176  reinterpret_cast<const std::int64_t *>(mat_a), reinterpret_cast<const std::int64_t *>(mat_b),
177  rows_a, cols_a, cols_b);
178  break;
179 
180  case hebench::APIBridge::DataType::Float32:
181  matMul<float>(reinterpret_cast<float *>(mat_result),
182  reinterpret_cast<const float *>(mat_a), reinterpret_cast<const float *>(mat_b),
183  rows_a, cols_a, cols_b);
184  break;
185 
186  case hebench::APIBridge::DataType::Float64:
187  matMul<double>(reinterpret_cast<double *>(mat_result),
188  reinterpret_cast<const double *>(mat_a), reinterpret_cast<const double *>(mat_b),
189  rows_a, cols_a, cols_b);
190  break;
191 
192  default:
193  throw std::invalid_argument(IL_LOG_MSG_CLASS("Unknown data type."));
194  break;
195  } // end switch
196 }

212 void DataGenerator::init(std::uint64_t rows_a, std::uint64_t cols_a, std::uint64_t cols_b,
213  std::uint64_t batch_size_mat_a,
214  std::uint64_t batch_size_mat_b,
215  hebench::APIBridge::DataType data_type)
216 {
217  // Load/generate and initialize the data for matrix multiplication:
218  // M2 = M0 * M1
219 
220  // number of samples in each input parameter and output
221  std::size_t batch_sizes[InputDim0 + 1] = {
222  batch_size_mat_a,
223  batch_size_mat_b,
224  batch_size_mat_a * batch_size_mat_b
225  };
226  // initialize base data (data packs)
227  PartialDataLoader::init(InputDim0, batch_sizes, OutputDim0);
228 
229  // store the dimensions of each matrix
230  std::array<std::pair<std::uint64_t, std::uint64_t>, InputDim0 + 1> mat_dims; // rows <=> first, cols <=> second
231  mat_dims[0] = std::make_pair(rows_a, cols_a);
232  mat_dims[1] = std::make_pair(cols_a, cols_b);
233  mat_dims[2] = std::make_pair(rows_a, cols_b);
234 
235  // compute buffer size for each matrix
236  std::uint64_t buffer_sizes[InputDim0 + 1];
237  for (std::size_t i = 0; i < InputDim0 + 1; ++i)
238  {
239  buffer_sizes[i] = mat_dims[i].first * mat_dims[i].second * PartialDataLoader::sizeOf(data_type);
240  } // end for
241 
242  // allocate memory for each matrix buffer
243  allocate(buffer_sizes, // sizes (in bytes) for each input matrix
244  InputDim0, // number of input matrices
245  buffer_sizes + InputDim0, // sizes (in bytes) for each output matrix
246  OutputDim0); // number of output matrices
247 
248  // at this point all NativeDataBuffers have been allocated and pointed to the correct locations
249 
250  // fill up the matrices data
251 
252  // input
253  for (std::size_t mat_i = 0; mat_i < InputDim0; ++mat_i)
254  {
255  for (std::uint64_t i = 0; i < batch_sizes[mat_i]; ++i)
256  {
257  // generate the data
258  DataGeneratorHelper::generateRandomMatrixN(data_type,
259  getParameterData(mat_i).p_buffers[i].p,
260  mat_dims[mat_i].first, // rows
261  mat_dims[mat_i].second, // columns
262  0.0, 10.0);
263  } // end for
264  } // end for
265 
266  // output
267  //#pragma omp parallel for collapse(2)
268  for (std::uint64_t m0_i = 0; m0_i < batch_sizes[0]; ++m0_i)
269  {
270  for (std::uint64_t m1_i = 0; m1_i < batch_sizes[1]; ++m1_i)
271  {
272  // find the index for the result buffer based on the input indices
273  std::uint64_t ppi[] = { m0_i, m1_i };
274  std::uint64_t r_i = getResultIndex(ppi);
275 
276  // generate the data
277  DataGeneratorHelper::matMul(data_type,
278  getResultData(0).p_buffers[r_i].p,
279  getParameterData(0).p_buffers[m0_i].p,
280  getParameterData(1).p_buffers[m1_i].p,
281  mat_dims[0].first, mat_dims[0].second, // dims for m0
282  mat_dims[1].second); // dims for m1
283  } // end for
284  } // end for
285 
286  // all data has been generated at this point
287 }
