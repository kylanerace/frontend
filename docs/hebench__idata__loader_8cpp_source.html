157 void PartialDataLoader::init(std::size_t input_dim,
158  const std::size_t *input_count_per_dim,
159  std::size_t output_dim)
160 {
161  if (input_dim <= 0)
162  throw std::invalid_argument(IL_LOG_MSG_CLASS("Invalid input dimensions: 'input_dim' must be positive."));
163  if (output_dim <= 0)
164  throw std::invalid_argument(IL_LOG_MSG_CLASS("Invalid output dimensions: 'output_dim' must be positive."));
165 
166  m_input_data.resize(input_dim);
167  m_output_data.resize(output_dim);
168 
169  std::size_t output_count_per_dim = 1;
170  for (std::size_t i = 0; i < m_input_data.size(); ++i)
171  {
172  if (input_count_per_dim[i] <= 0)
173  {
174  std::stringstream ss;
175  ss << "Invalid batch size for dimension " << i << ": 'input_count_per_dim[" << i << "]' must be positive.";
176  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
177  } // end if
178 
179  output_count_per_dim *= input_count_per_dim[i];
180  m_input_data[i] = createDataPack(input_count_per_dim[i], i);
181  } // end for
182 
183  for (std::size_t i = 0; i < m_output_data.size(); ++i)
184  {
185  m_output_data[i] = createDataPack(output_count_per_dim, i);
186  } // end for
187 }

189 void PartialDataLoader::allocate(const std::uint64_t *input_buffer_sizes,
190  std::size_t input_buffer_sizes_count,
191  const std::uint64_t *output_buffer_sizes,
192  std::size_t output_buffer_sizes_count,
193  bool allocate_output)
194 {
195  // allocate data for parameters and results
196 
197  // cache batch sizes
198  std::vector<std::uint64_t> input_batch_sizes(m_input_data.size());
199  for (std::size_t i = 0; i < m_input_data.size(); ++i)
200  input_batch_sizes[i] = m_input_data[i]->buffer_count;
201  std::vector<std::uint64_t> output_batch_sizes(m_output_data.size());
202  for (std::size_t i = 0; i < m_output_data.size(); ++i)
203  output_batch_sizes[i] = m_output_data[i]->buffer_count;
204 
205  if (input_buffer_sizes_count < input_batch_sizes.size())
206  {
207  std::stringstream ss;
208  ss << "Invalid number of input buffers 'input_buffer_sizes_count'. Expected, at least "
209  << input_batch_sizes.size() << ", but " << input_buffer_sizes_count << " received.";
210  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
211  } // end if
212  if (output_buffer_sizes_count < output_batch_sizes.size())
213  {
214  std::stringstream ss;
215  ss << "Invalid number of output buffers 'output_buffer_sizes_count'. Expected, at least "
216  << output_batch_sizes.size() << ", but " << output_buffer_sizes_count << " received.";
217  throw std::invalid_argument(IL_LOG_MSG_CLASS(ss.str()));
218  } // end if
219 
220  // compute total space required for the data buffers
221  std::uint64_t total_raw_size = 0;
222  std::uint64_t output_start;
223  for (std::size_t i = 0; i < input_batch_sizes.size(); ++i)
224  total_raw_size += input_buffer_sizes[i] * input_batch_sizes[i];
225  output_start = total_raw_size;
226  for (std::size_t i = 0; i < output_batch_sizes.size(); ++i)
227  total_raw_size += output_buffer_sizes[i] * output_batch_sizes[i];
228  // allocate space for the data buffers
229  m_raw_buffer.resize(total_raw_size, 0);
230 
231  // m_raw_buffer has a chunk of memory allocated enough to hold the all the data and
232  // now, we have to point the NativeDataBuffers for each sample to the right locations:
233 
234  // param[0, 0] param[0, 1] param[0, 2] param[1, 0] param[1, 1] param[1, 2] ...
235  // v v v v v v
236  // [ ***********************************************************************************... ]
237 
238  // initialize the buffers
239 
240  // point parameter start buffers to correct start location
241  std::vector<std::uint8_t *> input_buffers(input_batch_sizes.size(), m_raw_buffer.data());
242  for (std::uint64_t i = 1; i < input_batch_sizes.size(); ++i)
243  input_buffers[i] += i * input_buffer_sizes[i - 1] * input_batch_sizes[i - 1];
244  if (!input_buffers.empty())
245  // make sure we are pointing to the right location in the master buffer
246  assert(input_buffers.front() == m_raw_buffer.data());
247  std::vector<std::uint8_t *> output_buffers;
248  if (allocate_output)
249  {
250  output_buffers.resize(output_batch_sizes.size(), m_raw_buffer.data() + output_start);
251  for (std::uint64_t i = 1; i < output_batch_sizes.size(); ++i)
252  output_buffers[i] += i * output_buffer_sizes[i - 1] * output_batch_sizes[i - 1];
253  if (!output_buffers.empty())
254  // make sure we are pointing to the right location in the master buffer
255  assert(output_buffers.front() == m_raw_buffer.data() + output_start);
256  } // end if
257 
258  // point the NativeDataBuffers for each data sample to the correct memory location
259 
260  // input
261  for (std::size_t param_i = 0; param_i < m_input_data.size(); ++param_i)
262  {
263  //#pragma omp parallel for
264  for (std::uint64_t i = 0; i < input_batch_sizes[param_i]; ++i)
265  {
266  // point to the start of the data in the raw data buffer
267  m_input_data[param_i]->p_buffers[i].p = input_buffers[param_i] + i * input_buffer_sizes[param_i];
268  m_input_data[param_i]->p_buffers[i].size = input_buffer_sizes[param_i];
269  m_input_data[param_i]->p_buffers[i].tag = 0;
270  } // end for
271  } // end for
272  // output
273  for (std::size_t output_i = 0; output_i < m_output_data.size(); ++output_i)
274  {
275  //#pragma omp parallel for
276  for (std::uint64_t i = 0; i < output_batch_sizes[output_i]; ++i)
277  {
278  // point to the start of the data in the raw data buffer
279  m_output_data[output_i]->p_buffers[i].p = allocate_output ?
280  output_buffers[output_i] + i * output_buffer_sizes[output_i] :
281  nullptr;
282  m_output_data[output_i]->p_buffers[i].size = output_buffer_sizes[output_i];
283  m_output_data[output_i]->p_buffers[i].tag = 0;
284  } // end for
285  } // end for
286 
287  // all data has been allocated and pointed to at this point
288 }

323 std::uint64_t PartialDataLoader::getResultIndex(const std::uint64_t *param_data_pack_indices)
324 {
325  if (!param_data_pack_indices)
326  throw std::invalid_argument(IL_LOG_MSG_CLASS("Invalid null argument 'param_data_pack_indices'."));
327 
328  std::uint64_t retval = getParameterCount() > 0 ?
329  param_data_pack_indices[0] :
330  0;
331 
332  for (std::size_t param_i = 1; param_i < getParameterCount(); ++param_i)
333  {
334  assert(getParameterData(param_i).param_position == param_i);
335  if (param_data_pack_indices[param_i] >= getParameterData(param_i).buffer_count)
336  throw std::out_of_range(IL_LOG_MSG_CLASS("Index out of range: 'param_data_pack_indices['" + std::to_string(param_i) + "] == " + std::to_string(param_data_pack_indices[param_i]) + ". Expected value less than " + std::to_string(getParameterData(param_i).buffer_count) + "."));
337  retval = param_data_pack_indices[param_i] + getParameterData(param_i).buffer_count * retval;
338  } // end for
339 
340  return retval;
